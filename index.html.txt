<script>
    function toggleSettings() {
        const panel = document.getElementById('settingsPanel');
        panel.style.display = panel.style.display === 'block' ? 'none' : 'block';
    }

    function calculate() {
        // 1. Вводные данные
        const area = parseFloat(document.getElementById('area').value);
        const temp = parseFloat(document.getElementById('temp').value);
        const complexity = parseFloat(document.getElementById('complexity').value);
        const type = document.getElementById('type').value;
        const system = document.getElementById('system').value;

        // 2. Коммерческие настройки
        const marginRate = parseFloat(document.getElementById('set_margin').value) / 100 + 1; // e.g. 1.25
        const minOrderPenalty = parseFloat(document.getElementById('set_minorder').value);
        const dayRate = parseFloat(document.getElementById('set_dayrate').value);
        const productivity = parseFloat(document.getElementById('set_prod').value); // m2 per day
        
        // Цены
        const priceBaseMat = parseFloat(document.getElementById('set_base').value);
        const priceVarnish = parseFloat(document.getElementById('set_varnish').value);
        const priceHydro = 10; // GEL
        const priceRetarder = 0.5; // Base price for retarder per m2

        // 3. Логика Объекта
        let techDays = 0;
        let layersVarnish = 2;
        let hasHydro = false;

        if (type === 'bathroom') { techDays = 11; layersVarnish = 3; hasHydro = true; } 
        else if (type === 'floor') { techDays = 7; layersVarnish = 3; hasHydro = false; } 
        else if (type === 'wall') { techDays = 5; layersVarnish = 2; hasHydro = false; }

        // 4. Логика Времени (Дни)
        // Реальные дни = Берем максимум между технологическими паузами и площадью поделенной на выработку
        let activeDays = Math.ceil(area / productivity);
        let realDays = Math.max(techDays, activeDays);

        // LAB BETON штраф к работе (+25% к ФОТ мастера)
        let systemLaborMultiplier = (system === 'lab') ? 1.25 : 1.0;

        // Расчет ФОТ (С учетом дней, сложности стен и штрафа системы)
        let totalLabor = (realDays * dayRate) * complexity * systemLaborMultiplier;

        // 5. Логика Материалов
        // Физика лака (первый слой впитывается больше, 3 слоя = коэфф 1.15)
        let varnishFactor = (layersVarnish === 3) ? 1.15 : 1.1;
        let totalVarnishCost = layersVarnish * priceVarnish * area * varnishFactor;

        // Температурный градиент
        let retarderTotal = 0;
        const warningBox = document.getElementById('tempWarning');
        if (temp > 22) {
            let tempFactor = (temp - 22) / 10; // +10% за каждый градус свыше 22
            retarderTotal = area * priceRetarder * (1 + tempFactor);
            warningBox.style.display = 'block';
        } else {
            warningBox.style.display = 'none';
        }

        let totalMat = (priceBaseMat * area) + totalVarnishCost + retarderTotal;

        // Гидроизоляция
        if (hasHydro) {
            totalMat += (priceHydro * 0.5) * area;   // 50% мат
            totalLabor += (priceHydro * 0.5) * area; // 50% работа
        }

        // 6. Итоговая Коммерция
        let baseCost = totalMat + totalLabor;
        let finalCost = baseCost * marginRate;

        // Штраф за мелкий объект
        if (area < 10) {
            finalCost += minOrderPenalty;
        }

        let profit = finalCost - baseCost;
        let perM2 = finalCost / area;

        // 7. Расчет закупки (Физика объемов) - с запасом 10%
        let kgBase = Math.ceil(area * 1.5 * 1.1);
        let kgFin = Math.ceil(area * 0.5 * 1.1);
        let literLatex = Math.ceil(area * 0.65 * 1.1);
        let literVarnish = (area * 0.1 * layersVarnish * varnishFactor * 1.1).toFixed(1);

        // 8. Вывод в UI
        document.getElementById('statsGrid').style.display = 'grid';
        document.getElementById('matBox').style.display = 'block';

        document.getElementById('out_days').innerText = realDays;
        document.getElementById('out_matcost').innerText = Math.round(totalMat) + " ₾";
        document.getElementById('out_laborcost').innerText = Math.round(totalLabor) + " ₾";
        document.getElementById('out_margin').innerText = Math.round(profit) + " ₾";

        document.getElementById('mat_base_kg').innerText = kgBase + " кг";
        document.getElementById('mat_fin_kg').innerText = kgFin + " кг";
        document.getElementById('mat_latex_l').innerText = literLatex + " л";
        document.getElementById('mat_varnish_l').innerText = literVarnish + " л";

        document.getElementById('final_total').innerText = Math.round(finalCost) + " ₾";
        document.getElementById('final_m2').innerText = Math.round(perM2) + " ₾ / м²";
    }
</script>
